import numpy as np
import matplotlib.pyplot as plt
import time

# ================================
# SEKHEM CC — REAL-TIME ENGINE
# Founder: Mohamed Abd Elnaby
# ================================

dt = 0.02  # زمن حقيقي
N_SEKHEM = 3

# الحالة
X_SEKHEM = np.zeros(N_SEKHEM)
Y_target = 8.0

# معاملات
lambda_SEKHEM = 5.0
K_SEKHEM = 7.0
gamma_SEKHEM = 1.2

# رسم مباشر
plt.ion()
fig, ax = plt.subplots()

while True:

    # 🔥 Target (حركة حقيقية مستمرة)
    t_real = time.time()
    Y_target = 5 + 3*np.sin(t_real)

    X_new = X_SEKHEM.copy()

    for i in range(N_SEKHEM):

        # ================================
        # SEKHEM TRIAD
        # ================================
        R_SEKHEM = Y_target + np.random.normal(0, 0.1)
        C_SEKHEM = X_SEKHEM[i]
        P_SEKHEM = Y_target + 0.2*np.cos(t_real)

        triad_mean_SEKHEM = (R_SEKHEM + C_SEKHEM + P_SEKHEM)/3

        # PAC
        Phi_SEKHEM_PAC = -gamma_SEKHEM*(X_SEKHEM[i] - triad_mean_SEKHEM)

        # Coupling بسيط
        Coupling_SEKHEM = np.mean(X_SEKHEM) - X_SEKHEM[i]

        # Dynamics
        dX_SEKHEM = (
            -lambda_SEKHEM*(X_SEKHEM[i] - triad_mean_SEKHEM)
            + Phi_SEKHEM_PAC
            + Coupling_SEKHEM
            - K_SEKHEM*(X_SEKHEM[i] - Y_target)
        )

        X_new[i] += dt * dX_SEKHEM

    X_SEKHEM = X_new

    # ================================
    # PLOT REAL-TIME
    # ================================
    ax.clear()
    ax.scatter(range(N_SEKHEM), X_SEKHEM, label="SEKHEM Agents")
    ax.scatter(range(N_SEKHEM), [Y_target]*N_SEKHEM, color='red', label="Target")

    ax.set_ylim(0, 10)
    ax.set_title("SEKHEM CC — REAL-TIME INTERCEPTION")
    ax.legend()

    plt.pause(0.01)

    # ⏱️ Real-time sync
    time.sleep(dt)

import numpy as np
import matplotlib.pyplot as plt
import time

# ================================
# SEKHEM CC — 2D REAL INTERCEPTION
# Founder: Mohamed Abd Elnaby
# ================================

dt = 0.02

# حالة الهدف (position + velocity)
Y = np.array([8.0, 8.0])
VY = np.array([-1.0, -0.5])

# حالة النظام
X = np.array([0.0, 0.0])
VX = np.array([0.0, 0.0])

# معاملات SEKHEM
lambda_SEKHEM = 6.0
K_SEKHEM = 8.0
gamma_SEKHEM = 1.2

trajectory_X = []
trajectory_Y = []

start_time = time.time()

while True:

    t_real = time.time() - start_time

    # ================================
    # Target Maneuver (real physics)
    # ================================
    VY += np.array([0.2*np.sin(t_real), 0.2*np.cos(t_real)]) * dt
    Y += VY * dt

    # ================================
    # SEKHEM TRIAD
    # ================================
    R = Y + np.random.normal(0, 0.1, 2)
    C = X
    P = Y + VY * 0.3  # prediction forward

    triad_mean = (R + C + P) / 3

    # ================================
    # PAC
    # ================================
    PAC = -gamma_SEKHEM * (X - triad_mean)

    # ================================
    # SEKHEM FORCE
    # ================================
    F = (
        -lambda_SEKHEM * (X - triad_mean)
        + PAC
        - K_SEKHEM * (X - Y)
    )

    # Physics (Newton)
    VX += F * dt
    X += VX * dt

    # تخزين
    trajectory_X.append(X.copy())
    trajectory_Y.append(Y.copy())

    # ================================
    # رسم
    # ================================
    plt.clf()
    trajX = np.array(trajectory_X)
    trajY = np.array(trajectory_Y)

    plt.plot(trajX[:,0], trajX[:,1], label="SEKHEM")
    plt.plot(trajY[:,0], trajY[:,1], label="Target")

    plt.scatter(X[0], X[1])
    plt.scatter(Y[0], Y[1], color='red')

    plt.legend()
    plt.title("SEKHEM 2D Real Interception")

    plt.xlim(-2, 10)
    plt.ylim(-2, 10)

    plt.pause(0.01)

    # ================================
    # INTERCEPTION CHECK
    # ================================
    error = np.linalg.norm(X - Y)

    if error < 0.2:
        print("🔥 INTERCEPTED!")
        print("⏱ Time:", round(t_real, 2), "seconds")
        break

    time.sleep(dt)
